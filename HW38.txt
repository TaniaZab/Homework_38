package com.company;

public class Main {

    public static void main(String[] args) {
	// определяем двумерный массив из 5 строк и 8 столбцов
        int [][] Array_2 = new int[5][8];
        //организовываем цикл прохождения каждой строки массива от i=0 до Array_2.length=4
        for(int i =0; i<Array_2.length; i++){
            //организовываем цикл прохождения каждого столбца в выбранной строке от j=0 до Array_2[i].length=7
            for(int j=0; j<Array_2[i].length; j++){
                //генерируем целые случайные числа от -99 до +99. Учитывая, что Math.random() принимает значения от [0...1)
                //умножаем на 199 (значения составят от [0...199)) и вычитаем 99 (значения составят [-99...100)).
                //Число 100 не включается задаваемый диапазон, следовательно ближайшее целое число составит 99.
                //Массив будт заполняться целыми числами [-99...99]
                Array_2[i][j] = (int) ((Math.random()*199)-99);
                //выводим на экран каждое число массива
                System.out.print(Array_2[i][j] + " ");
            }
            //после вывода на экран чисел в столбцах i-й строки переводим курсор на новую сттроку
            System.out.println();
        }
        //предполагаем, что максимальное число (max) равно первому элементу в массиве Array_2[0][0]
        int max = Array_2[0][0];
        //организовываем цикл (переменная i) прохождения каждой строки массива
        for(int i = 0; i<Array_2.length; i++){
            //организовываем цикл (переменная j) прохождения каждого столбца в i-й строке
            for(int j=0; j<Array_2[i].length; j++){
                //сравниваем каждое число моссива со значением max. Если элемент массива Array_2[i][j] больше max, то
                //переменной max присваиваем текущее значение числа массива Array_2[i][j]
                if (Array_2[i][j] > max)  max=Array_2[i][j];
            }
        }
        System.out.println("Максимальное значение элемента массива составляет "+max);
    }
}
